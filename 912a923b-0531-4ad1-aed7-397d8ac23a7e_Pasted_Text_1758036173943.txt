        function selectService(element, serviceName) {
            // Убрать выделение со всех услуг
            document.querySelectorAll('.service-card').forEach(service => {

            });

            document.getElementById('confirmation-service-name').textContent = serviceName;
            
            // Установить цену в зависимости от услуги
            let price = '0 ₽';
            if (serviceName === 'Теннис') {
                price = '1200 ₽';
            } else if (serviceName === 'Персональная тренировка') {
                price = '2500 ₽';
            }
            
            document.getElementById('summary-service-price').textContent = price;
            
            // Пересчитать итоговую сумму
            const peakSurcharge = serviceName === 'Теннис' ? 240 : 0;
            const totalPrice = serviceName === 'Теннис' ? 1440 : 
                              serviceName === 'Персональная тренировка' ? 2500 : 0;
            
            document.getElementById('total-price').textContent = `${totalPrice} ₽`;
        }

        // Получить следующий час
        function getNextHour(time) {
            const [hours, minutes] = time.split(':');
            const nextHour = parseInt(hours) + 1;
            return `${nextHour.toString().padStart(2, '0')}:${minutes}`;
        }

            element.classList.add('selected');

        // Отправка сообщения в чат
        function sendMessage() {
            const input = document.getElementById('message-input');
            const message = input.value.trim();
            
            if (message) {
                const messagesContainer = document.getElementById('chat-messages');
                const messageElement = document.createElement('div');
                messageElement.className = 'message user';
                messageElement.innerHTML = `<strong>Вы:</strong><br>${message}`;
                messagesContainer.appendChild(messageElement);
                
                // Ответ от поддержки
                setTimeout(() => {
                    const responseElement = document.createElement('div');
                    responseElement.className = 'message support';
                    responseElement.innerHTML = '<strong>Поддержка:</strong><br>Спасибо за ваше сообщение! Мы ответим в ближайшее время.';
                    messagesContainer.appendChild(responseElement);
                    messagesContainer.scrollTop = messagesContainer.scrollHeight;
                }, 1000);
                
                input.value = '';
                messagesContainer.scrollTop = messagesContainer.scrollHeight;
            }
        }

            const content = document.getElementById('history-content');

            content.classList.toggle('active');

                icon.className = 'fas fa-chevron-down';

            if (e.touches.length > 1) {
                e.preventDefault();
            }
        }, { passive: false });

        let lastTouchEnd = 0;
        document.addEventListener('touchend', function(e) {
            const now = (new Date()).getTime();
            if (now - lastTouchEnd <= 300) {
                e.preventDefault();
            }
            lastTouchEnd = now;
        }, false);
